```csharp  
   public static implicit operator int(iii x)

   }

   }

```  
  
## <a name="example"></a><span data-ttu-id="57386-109">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="57386-109">Example</span></span>  

 <span data-ttu-id="57386-110">El ejemplo siguiente genera la advertencia CS0559.</span><span class="sxs-lookup"><span data-stu-id="57386-110">The following sample generates CS0559.</span></span>  
  
```csharp  
// CS0559_b.cs  
// compile with: /target:library  
public struct S  
{  
   public static S operator ++(S? s) { return new S(); }   // CS0559  
   public static S operator --(S? s) { return new S(); }   // CS0559  
}  
  
public struct T  
{  
// OK  
   public static T operator --(T t) { return new T(); }  
   public static T operator ++(T t) { return new T(); }  
  
   public static T? operator --(T? t) { return new T(); }  
   public static T? operator ++(T? t) { return new T(); }  
}  
```
