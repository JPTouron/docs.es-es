## <a name="example-1"></a><span data-ttu-id="cc31d-110">Ejemplo 1</span><span class="sxs-lookup"><span data-stu-id="cc31d-110">Example 1</span></span>

        var a = 3, b = 2; // CS0819

        // First correction option.
